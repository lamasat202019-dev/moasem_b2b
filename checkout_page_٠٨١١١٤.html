    <script>
        // بيانات الطلب
        let orderData = {
            shippingMethod: 'standard',
            paymentMethod: 'card',
            shippingCost: 0,
            discount: 25,
            subtotal: 375,
            tax: 18.75,
            total: 368.75
        };
       
        // تهيئة التطبيق
        document.addEventListener('DOMContentLoaded', function() {
            // تحديث الخطوات
            updateCheckoutSteps();
           
            // تحميل بيانات السلة
            loadCartData();
           
            // تحديث المبلغ الإجمالي
            updateOrderSummary();
        });
       
        // تحديث خطوات الدفع
        function updateCheckoutSteps() {
            const steps = document.querySelectorAll('.step');
            steps.forEach((step, index) => {
                step.classList.remove('active', 'completed');
                if (index < 1) {
                    step.classList.add('completed');
                } else if (index === 1) {
                    step.classList.add('active');
                }
            });
        }
       
        // تحميل بيانات السلة
        function loadCartData() {
            // يمكن جلب بيانات السلة من localStorage أو API
            const cart = JSON.parse(localStorage.getItem('mawsem_cart') || '[]');
            if (cart.length === 0) {
                // إذا كانت السلة فارغة، عرض رسالة
                showAlert('سلة التسوق فارغة', 'warning');
                setTimeout(() => {
                    window.location.href = 'index.html';
                }, 2000);
            }
        }
       
        // تحديث الكمية
        function updateQuantity(itemId, change) {
            const quantityElement = document.getElementById('quantity' + itemId);
            let quantity = parseInt(quantityElement.textContent);
            quantity += change;
           
            if (quantity < 1) {
                showAlert('الكمية لا يمكن أن تكون أقل من 1', 'warning');
                return;
            }
           
            quantityElement.textContent = quantity;
           
            // تحديث السعر
            updateItemPrice(itemId, quantity);
            updateOrderSummary();
        }
       
        // تحديث سعر المنتج
        function updateItemPrice(itemId, quantity) {
            const prices = {
                1: 120,
                2: 45
            };
           
            const price = prices[itemId];
            const total = price * quantity;
           
            // تحديث العرض في الواجهة
            const itemElement = document.querySelector(`#quantity${itemId}`).closest('.cart-item');
            const totalElement = itemElement.querySelector('.cart-item-quantity span:last-child');
            totalElement.textContent = total + ' ر.س';
           
            // تحديث المجموع الجزئي
            orderData.subtotal = 0;
            document.querySelectorAll('.cart-item').forEach(item => {
                const itemTotal = parseInt(item.querySelector('.cart-item-quantity span:last-child').textContent);
                orderData.subtotal += itemTotal;
            });
        }
       
        // تحديث ملخص الطلب
        function updateOrderSummary() {
            // تحديث الشحن
            document.getElementById('shippingCost').textContent =
                orderData.shippingCost === 0 ? 'مجاناً' : orderData.shippingCost + ' ر.س';
           
            // تحديث الضرائب (5% من المجموع الجزئي)
            orderData.tax = orderData.subtotal * 0.05;
           
            // حساب المجموع النهائي
            orderData.total = orderData.subtotal + orderData.shippingCost + orderData.tax - orderData.discount;
           
            // تحديث الواجهة
            document.getElementById('totalAmount').textContent = orderData.total.toFixed(2) + ' ر.س';
           
            // تحديث العناصر الأخرى في الملخص
            document.querySelectorAll('.summary-row')[0].children[1].textContent = orderData.subtotal + ' ر.س';
            document.querySelectorAll('.summary-row')[3].children[1].textContent = orderData.tax.toFixed(2) + ' ر.س';
        }
       
        // اختيار طريقة الشحن
        function selectShipping(method) {
            orderData.shippingMethod = method;
           
            // تحديث الواجهة
            document.querySelectorAll('.shipping-method').forEach(el => {
                el.classList.remove('selected');
            });
            event.target.closest('.shipping-method').classList.add('selected');
           
            // تحديث تكلفة الشحن
            const shippingCosts = {
                standard: 0,
                express: 45,
                nextday: 85
            };
           
            orderData.shippingCost = shippingCosts[method];
            updateOrderSummary();
        }
       
        // اختيار طريقة الدفع
        function selectPayment(method) {
            orderData.paymentMethod = method;
           
            // تحديث الواجهة
            document.querySelectorAll('.payment-method').forEach(el => {
                el.classList.remove('selected');
            });
            event.target.closest('.payment-method').classList.add('selected');
           
            // إذا كانت طريقة الدفع عند الاستلام، أضف رسوم
            if (method === 'cod') {
                orderData.shippingCost += 15;
                updateOrderSummary();
            }
        }
       
        // تطبيق الخصم
        function applyDiscount() {
            const code = document.getElementById('discountCode').value.trim();
            const discountCodes = {
                'MAWSEM10': 10,
                'WELCOME25': 25,
                'SAVE50': 50
            };
           
            if (code in discountCodes) {
                orderData.discount = discountCodes[code];
                updateOrderSummary();
                showAlert(`تم تطبيق الخصم بنجاح! وفرت ${orderData.discount} ر.س`, 'success');
            } else {
                showAlert('كود الخصم غير صحيح أو منتهي الصلاحية', 'error');
            }
        }
       
        // إتمام الطلب
        function completeOrder() {
            // التحقق من الموافقة على الشروط
            if (!document.getElementById('agreeTerms').checked) {
                showAlert('يجب الموافقة على الشروط والأحكام', 'error');
                return;
            }
           
            // التحقق من طريقة الدفع
            if (!orderData.paymentMethod) {
                showAlert('يرجى اختيار طريقة الدفع', 'error');
                document.getElementById('paymentSection').scrollIntoView({ behavior: 'smooth' });
                return;
            }
           
            // محاكاة عملية الدفع
            showAlert('جاري معالجة طلبك...', 'info');
           
            setTimeout(() => {
                // مسح السلة
                localStorage.removeItem('mawsem_cart');
               
                // عرض نافذة النجاح
                document.getElementById('successModal').classList.add('active');
               
                // تحديث عداد السلة
                updateCartBadge(0);
            }, 2000);
        }

        // إغلاق نافذة إضافة عنوان
        function closeAddAddressModal() {
            document.getElementById('addAddressModal').classList.remove('active');
            document.getElementById('addressForm').reset();
        }
       
        // حفظ العنوان
        function saveAddress() {
            const form = document.getElementById('addressForm');
            const formData = new FormData(form);
           
            // التحقق من المدخلات
            let valid = true;
            form.querySelectorAll('[required]').forEach(input => {
                if (!input.value.trim()) {
                    valid = false;
                    input.style.borderColor = 'var(--danger-color)';
                } else {
                    input.style.borderColor = '';
                }
            });
           
            if (!valid) {
                showAlert('الرجاء ملء جميع الحقول المطلوبة', 'error');
                return;
            }
           
            closeAddAddressModal();
            showAlert('تم إضافة العنوان بنجاح', 'success');
        }

        function trackOrder() {
            document.getElementById('successModal').classList.remove('active');
            window.location.href = 'order_tracking.html?order=MAW-78901';
        }
       
        // مواصلة التسوق
        function continueShopping() {
            document.getElementById('successModal').classList.remove('active');
            window.location.href = 'index.html';
        }
       
        // عرض تنبيه
        function showAlert(message, type = 'info') {
            const alert = document.createElement('div');
            alert.className = `alert alert-${type === 'error' ? 'danger' : type}`;
            alert.innerHTML = `
                <i class="fas fa-${type === 'success' ? 'check-circle' : type === 'error' ? 'exclamation-circle' : 'info-circle'}"></i>
                ${message}
            `;
           
            const container = document.querySelector('.main-content');
            container.insertBefore(alert, container.firstChild);
           
            setTimeout(() => {
                alert.remove();
            }, 3000);
        }
