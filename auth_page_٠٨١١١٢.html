        function loginUser() {
            const email = document.querySelector('#loginForm input[type="text"]').value;
            const password = document.getElementById('loginPassword').value;
            const userType = document.querySelector('input[name="loginType"]:checked').value;
            const rememberMe = document.getElementById('rememberMe').checked;
           
            // التحقق من المدخلات
            if (!email || !password) {
                showAlert('الرجاء ملء جميع الحقول', 'error');
                return;
            }
           
            // محاكاة تسجيل الدخول
            showAlert('جاري تسجيل الدخول...', 'info');
           
            setTimeout(() => {
                showAlert('تم تسجيل الدخول بنجاح!', 'success');
               
                // حفظ بيانات المستخدم
                const userData = {
                    email: email,
                    type: userType,
                    remember: rememberMe
                };
               
                localStorage.setItem('mawsem_user', JSON.stringify(userData));
               
                // التوجيه بناءً على نوع المستخدم
                setTimeout(() => {
                    if (userType === 'vendor') {
                        window.location.href = 'vendor_dashboard.html';
                    } else {
                        window.location.href = 'index.html';
                    }
                }, 1000);
            }, 1500);
        }
       
        // إنشاء حساب جديد
        function signupUser() {
            const companyName = document.querySelector('#signupForm input[placeholder*="اسم شركتك"]').value;
            const email = document.querySelector('#signupForm input[type="email"]').value;
            const phone = document.querySelector('#signupForm input[type="tel"]').value;
            const password = document.getElementById('signupPassword').value;
            const confirmPassword = document.getElementById('confirmPassword').value;
            const userType = document.querySelector('input[name="userType"]:checked').value;
            const acceptTerms = document.getElementById('acceptTerms').checked;
           
            // التحقق من المدخلات
            if (!companyName || !email || !phone || !password || !confirmPassword) {
                showAlert('الرجاء ملء جميع الحقول', 'error');
                return;
            }
           
            if (!acceptTerms) {
                showAlert('يجب الموافقة على الشروط والأحكام', 'error');
                return;
            }
           
            if (password !== confirmPassword) {
                showAlert('كلمات المرور غير متطابقة', 'error');
                return;
            }
           
            // محاكاة إنشاء الحساب
            showAlert('جاري إنشاء حسابك...', 'info');
           
            setTimeout(() => {
                showAlert('تم إنشاء حسابك بنجاح!', 'success');
               
                // حفظ بيانات المستخدم
                const userData = {
                    name: companyName,
                    email: email,
                    phone: phone,
                    type: userType,
                    joined: new Date().toISOString()
                };
               
                localStorage.setItem('mawsem_user', JSON.stringify(userData));
               
                // التوجيه بناءً على نوع المستخدم
                setTimeout(() => {
                    if (userType === 'vendor') {
                        showAlert('ستتم مراجعة حسابك من قبل الإدارة خلال 24 ساعة', 'info');
                        window.location.href = 'vendor_dashboard.html';
                    } else {
                        window.location.href = 'index.html';
                    }
                }, 1500);
            }, 2000);
        }

        // قوة كلمة المرور
        function checkPasswordStrength() {
            const password = document.getElementById('signupPassword').value;
            const strengthBar = document.getElementById('passwordStrengthBar');
            const reqLength = document.getElementById('reqLength');
            const reqUppercase = document.getElementById('reqUppercase');
            const reqNumber = document.getElementById('reqNumber');
           
            let strength = 0;
            let messages = [];
           
            // الطول
            if (password.length >= 8) {
                strength += 1;
                reqLength.classList.add('met');
                reqLength.innerHTML = '<i class="fas fa-check" style="color: #2ed573;"></i> 8 أحرف على الأقل';
            } else {
                reqLength.classList.remove('met');
                reqLength.innerHTML = '<i class="fas fa-times" style="color: #ff4757;"></i> 8 أحرف على الأقل';
            }
           
            // حرف كبير
            if (/[A-Z]/.test(password)) {
                strength += 1;
                reqUppercase.classList.add('met');
                reqUppercase.innerHTML = '<i class="fas fa-check" style="color: #2ed573;"></i> حرف كبير واحد على الأقل';
            } else {
                reqUppercase.classList.remove('met');
                reqUppercase.innerHTML = '<i class="fas fa-times" style="color: #ff4757;"></i> حرف كبير واحد على الأقل';
            }
           
            // رقم
            if (/\d/.test(password)) {
                strength += 1;
                reqNumber.classList.add('met');
                reqNumber.innerHTML = '<i class="fas fa-check" style="color: #2ed573;"></i> رقم واحد على الأقل';
            } else {
                reqNumber.classList.remove('met');
                reqNumber.innerHTML = '<i class="fas fa-times" style="color: #ff4757;"></i> رقم واحد على الأقل';
            }
           
            // تحديث شريط القوة
            if (password.length === 0) {
                strengthBar.className = 'password-strength-bar';
                strengthBar.style.width = '0%';
            } else if (strength === 1) {
                strengthBar.className = 'password-strength-bar weak';
            } else if (strength === 2) {
                strengthBar.className = 'password-strength-bar fair';
            } else if (strength === 3) {
                strengthBar.className = 'password-strength-bar strong';
            }
        }
       
        // التحقق من تطابق كلمات المرور
        function checkPasswordMatch() {
            const password = document.getElementById('signupPassword').value;
            const confirmPassword = document.getElementById('confirmPassword').value;
            const matchText = document.getElementById('passwordMatchText');
           
            if (confirmPassword.length === 0) {
                matchText.style.display = 'none';
                return;
            }
           
            if (password === confirmPassword) {
                matchText.style.display = 'block';
                matchText.innerHTML = '<i class="fas fa-check" style="color: #2ed573;"></i> كلمات المرور متطابقة';
                matchText.style.color = '#2ed573';
            } else {
                matchText.style.display = 'block';
                matchText.innerHTML = '<i class="fas fa-times" style="color: #ff4757;"></i> كلمات المرور غير متطابقة';
                matchText.style.color = '#ff4757';
            }
        }
