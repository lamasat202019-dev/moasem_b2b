    <!-- الجافاسكريبت -->
    <script>
        // بيانات التطبيق
        let cart = [];
        let cartCount = 2;
       
        // تهيئة التطبيق
        document.addEventListener('DOMContentLoaded', function() {
            // تحديث عداد السلة
            updateCartBadge();
           
            // التحكم في القائمة الجانبية
            document.getElementById('menuToggle')?.addEventListener('click', toggleMenu);
            document.getElementById('closeMenu')?.addEventListener('click', toggleMenu);
           
            // نموذج RFQ
            document.getElementById('rfqForm')?.addEventListener('submit', function(e) {
                e.preventDefault();
                const formData = new FormData(this);
                alert('شكراً لك! تم استلام طلب عرض السعر بنجاح. سنتواصل معك خلال 24 ساعة.');
                this.reset();
            });
           
            // إغلاق القائمة عند النقر خارجها
            document.addEventListener('click', function(e) {
                const mobileMenu = document.getElementById('mobileMenu');
                const menuToggle = document.getElementById('menuToggle');
               
                if (mobileMenu.style.display === 'block' &&
                    !mobileMenu.contains(e.target) &&
                    !menuToggle.contains(e.target)) {
                    mobileMenu.style.display = 'none';
                }
            });
        });
       
        // البحث
        function performSearch() {
            const searchTerm = document.getElementById('mainSearch')?.value.trim();
            if (searchTerm) {
                alert(`جاري البحث عن: "${searchTerm}"`);
                // هنا يمكن إضافة منطق البحث الحقيقي
            } else {
                alert('الرجاء إدخال مصطلح البحث');
            }
        }
       
        // إضافة إلى السلة
        function addToCart(productId) {
            const products = {
                1: { name: 'كرتون التمور الفاخرة', price: 120, quantity: 1 },
                2: { name: 'زيت زيتون بكر ممتاز', price: 45, quantity: 1 },
                3: { name: 'قهوة عربية خالصة', price: 85, quantity: 1 }
            };
           
            const product = products[productId];
            if (product) {
                cart.push(product);
                cartCount++;
                updateCartBadge();
               
                // عرض رسالة تأكيد
                showAlert(`تم إضافة "${product.name}" إلى السلة`, 'success');
               
                // تحديث عرض السلة
                updateCartDisplay();
            }
        }
       
        // تحديث عداد السلة
        function updateCartBadge() {
            const badge = document.querySelector('.bottom-nav .nav-badge');
            if (badge) {
                badge.textContent = cartCount;
            }
           
            const headerBadge = document.querySelector('.header-btn .badge');
            if (headerBadge) {
                headerBadge.textContent = cartCount;
            }
        }
       
        // عرض/إخفاء القائمة
        function toggleMenu() {
            const menu = document.getElementById('mobileMenu');
            if (menu.style.display === 'block') {
                menu.style.display = 'none';
            } else {
                menu.style.display = 'block';
            }
        }
       
        // عرض/إخفاء السلة
        function toggleCart() {
            const cartSidebar = document.getElementById('cartSidebar');
            if (cartSidebar.style.display === 'block') {
                cartSidebar.style.display = 'none';
            } else {
                cartSidebar.style.display = 'block';
                updateCartDisplay();
            }
        }
       
        // إغلاق السلة
        function closeCart() {
            document.getElementById('cartSidebar').style.display = 'none';
        }
       
        // تحديث عرض السلة
        function updateCartDisplay() {
            const cartItems = document.getElementById('cartItems');
            const cartTotal = document.getElementById('cartTotal');
           
            if (!cartItems) return;
           
            if (cart.length === 0) {
                cartItems.innerHTML = '<p style="text-align: center; color: var(--text-light); padding: 20px;">السلة فارغة</p>';
                cartTotal.textContent = '0 ر.س';
                return;
            }
           
            let html = '';
            let total = 0;
           
            cart.forEach((item, index) => {
                total += item.price * item.quantity;
                html += `
                    <div class="cart-item">
                        <div class="cart-item-info">
                            <h4>${item.name}</h4>
                            <p>${item.price} ر.س × ${item.quantity}</p>
                        </div>
                        <div class="cart-item-actions">
                            <button onclick="removeFromCart(${index})" class="btn btn-sm btn-danger">
                                <i class="fas fa-trash"></i>
                            </button>
                        </div>
                    </div>
                `;
            });
           
            cartItems.innerHTML = html;
            cartTotal.textContent = total + ' ر.س';
        }
       
        // إزالة من السلة
        function removeFromCart(index) {
            cart.splice(index, 1);
            cartCount--;
            updateCartBadge();
            updateCartDisplay();
            showAlert('تم إزالة المنتج من السلة', 'warning');
        }
       
        // عرض تنبيه
        function showAlert(message, type = 'info') {
            const alert = document.createElement('div');
            alert.className = `alert alert-${type}`;
            alert.innerHTML = `
                <i class="fas fa-${type === 'success' ? 'check-circle' : type === 'warning' ? 'exclamation-triangle' : 'info-circle'}"></i>
                ${message}
            `;
           
            document.querySelector('.main-content').prepend(alert);
           
            setTimeout(() => {
                alert.remove();
            }, 3000);
        }
       
        // تنبيهات الترحيب
        window.onload = function() {
            setTimeout(() => {
                if (!localStorage.getItem('welcomeShown')) {
                    showAlert('مرحباً بك في مواسم! ابدأ رحلتك التجارية معنا اليوم.', 'success');
                    localStorage.setItem('welcomeShown', 'true');
                }
            }, 1000);
        };
    </script>
